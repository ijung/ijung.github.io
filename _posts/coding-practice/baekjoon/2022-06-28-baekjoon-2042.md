# 문제
## [2042 구간 합 구하기](https://www.acmicpc.net/problem/2042)
## ![screencapture](/posts/coding-practice/baekjoon/screencapture-acmicpc-net-problem-2042.png)

# 풀이
- Long형 data를 최대 1,000,000개 합쳐야 하는 문제이다.
- 단순한 sum으로는 timeout이 발생한다.(아래 오답 부분 참고)
- 구간의 합을 구하기 위해 segment tree를 이용한다.
- 풀이 추가예정

# 답
## kotlin code: 오답 - (당연하게도)timeout
```kotlin
fun main() {
    val (n, m, k) = readln().split(' ').map { it.toInt() }
    val array = LongArray(n) { readln().toLong() }
    repeat(m + k) {
        val strArray= readln().split(' ')
        when(strArray[0].toInt()) {
            1 -> array[strArray[1].toInt()-1] = strArray[2].toLong()
            2 -> println((strArray[1].toInt()-1 until strArray[2].toInt()).sumOf { array[it] })
        }
    }
}
```

## kotlin code: 정답 - Segment Tree 사용
```kotlin
package baekjoon.q2042

fun main() {
    val (n, m, k) = readln().split(' ').map { it.toInt() }
    val arr = LongArray(n) { readln().toLong() }
    val tree = createTree(arr)
    repeat(m + k) {
        val strArray= readln().split(' ')
        when(strArray[0].toInt()) {
            1 -> {
                val index = strArray[1].toInt()-1
                update(0, n-1, 1, tree, index, strArray[2].toLong() - arr[index])
                arr[index] = strArray[2].toLong()
            }
            2 -> println(sum(0, n-1, strArray[1].toInt()-1, strArray[2].toInt()-1, 1, tree))
        }
    }
}

fun createTree(arr:LongArray) = LongArray(arr.getTreeSize()).let {
    initTree(0, arr.lastIndex, arr, 1, it)
    it
}

fun initTree(arrStartIndex: Int, arrEndIndex:Int, arr:LongArray, node:Int, tree:LongArray): Long {
    if(arrStartIndex == arrEndIndex) {
        tree[node] = arr[arrStartIndex]
        return tree[node]
    }
    val arrMidIndex = (arrStartIndex + arrEndIndex) / 2
    tree[node] = initTree(arrStartIndex, arrMidIndex, arr, node * 2, tree) + initTree(arrMidIndex + 1, arrEndIndex, arr,node * 2 + 1, tree)
    return tree[node]
}

private fun LongArray.getTreeSize() = Math.pow(2.0, kotlin.math.ceil(kotlin.math.log2(this.size.toDouble()))).toInt() * 2

fun sum(arrStartIndex:Int, arrEndIndex:Int, sumStartIndex:Int, sumEndIndex:Int, node:Int, tree:LongArray): Long {
    if(sumEndIndex < arrStartIndex || sumStartIndex > arrEndIndex) return 0L
    if(sumStartIndex <= arrStartIndex && arrEndIndex <= sumEndIndex) return tree[node]
    val arrMidIndex = (arrStartIndex + arrEndIndex) / 2
    return sum(arrStartIndex, arrMidIndex, sumStartIndex, sumEndIndex, node * 2, tree) + sum(arrMidIndex + 1, arrEndIndex, sumStartIndex, sumEndIndex, node * 2 + 1, tree)
}

fun update(arrStartIndex: Int, arrEndIndex: Int, node: Int, tree: LongArray, index: Int, diffValue: Long) {
    if(index < arrStartIndex || index > arrEndIndex) return
    tree[node] += diffValue
    if(arrStartIndex == arrEndIndex) return
    val arrMidIndex = (arrStartIndex + arrEndIndex) / 2
    update(arrStartIndex, arrMidIndex, node * 2, tree, index, diffValue)
    update(arrMidIndex + 1, arrEndIndex, node * 2 + 1, tree, index, diffValue)
}
```
